#include<stdio.h>
struct dept{
    char name[50];
    int deptid;
};
struct emp{
    int empid;
    char empname[50];
    struct dept d;
};
int main (){
    struct emp e;
    struct emp *ptr;
    ptr=&e;
    printf("Enter the employee id:");
    scanf("%d",&ptr->empid);
    printf("Enter the employee name:");
    scanf("%s",ptr->empname);
    printf("Enter the dept name :");
    scanf("%s",ptr->d.name);
    printf("Enter the dept id:");
    scanf("%d",&ptr->d.deptid);
    
    printf("----Details---");
    printf("Dept id:%d",ptr->d.deptid);
    printf("Dept name:%s",ptr->d.name);
    printf("emp id:%d",ptr->empid);
    printf("emp name:%s",ptr->empname);
}
